//simple stack example
type Stack = List[Int]
def push(x: Int) = State[Stack, Unit] {
  case xs => (x :: xs, ())
}
val pop = State[Stack, Int] {
  case x :: xs => (xs, x)
  case Nil     => sys.error("empty")
}

def flatMap[B](f: A => State[S, B]): State[S, B] = 
  State(s => {
    val (a, s1) = run(s)
    f(a).run(s1)
  })
